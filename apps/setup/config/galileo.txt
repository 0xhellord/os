DriverFiles += [
    "qrkhostb.drv",
];

BootDrivers += BootDriversX86;

    "qrkhostb.drv",
];

//
// The Galileo uses the second debug device found for debugging.
//

BootEntry["DebugDevice"] = 1;
BootEntry["KernelArguments"] += " ps.env=CONSOLE=/dev/tty/Slave2 ";

//
// Set the EFI boot manager name.
//

BootmefiCopy["Destination"] += EFI_DEFAULT_APP_X86;
